🏢 Senaryo 1: Kullanıcı Hesap Yönetimi

Problem: 100 yeni çalışan için manuel hesap oluşturma

Çözüm: PowerShell ile otomatik hesap oluşturma

# 1. CSV dosyasından kullanıcı bilgilerini oku
$users = Import-Csv "C:\NewUsers.csv"

# 2. Her kullanıcı için hesap oluştur
foreach ($user in $users) {
    try {
        # Kullanıcı hesabı oluştur
        New-ADUser -Name $user.FullName -SamAccountName $user.Username -UserPrincipalName $user.Email -Path "OU=Users,DC=company,DC=com" -AccountPassword (ConvertTo-SecureString $user.Password -AsPlainText -Force) -Enabled $true
        
        # Grup üyeliği ekle
        Add-ADGroupMember -Identity "Domain Users" -Members $user.Username
        
        # E-posta kutusu oluştur
        Enable-Mailbox -Identity $user.Email -Database "Mailbox Database"
        
        Write-Log -Message "Kullanıcı oluşturuldu: $($user.Username)" -Level "INFO" -Path "C:\Logs\UserCreation.log"
    }
    catch {
        Write-Log -Message "Hata: $($_.Exception.Message)" -Level "ERROR" -Path "C:\Logs\UserCreation.log"
    }
}

🏢 Senaryo 2: Sistem Sağlık Kontrolü

Problem: 50 sunucunun sağlık durumunu manuel kontrol

Çözüm: PowerShell ile otomatik sağlık raporu

# 1. Sunucu listesi
$servers = @("Server1", "Server2", "Server3", "Server4", "Server5")

# 2. Sağlık kontrolü fonksiyonu
function Get-SystemHealth {
    param([string]$ComputerName)
    
    $health = @{
        ComputerName = $ComputerName
        Timestamp = Get-Date
        CPU = Get-WmiObject -Class Win32_Processor -ComputerName $ComputerName | Measure-Object -Property LoadPercentage -Average | Select-Object -ExpandProperty Average
        Memory = Get-WmiObject -Class Win32_OperatingSystem -ComputerName $ComputerName | Select-Object @{Name="FreeMemoryGB";Expression={[math]::Round($_.FreePhysicalMemory/1MB,2)}}
        Disk = Get-WmiObject -Class Win32_LogicalDisk -ComputerName $ComputerName | Where-Object {$_.DriveType -eq 3} | Select-Object DeviceID, @{Name="FreeSpaceGB";Expression={[math]::Round($_.FreeSpace/1GB,2)}}
    }
    
    return $health
}

# 3. Tüm sunucuları kontrol et
$healthReport = @()
foreach ($server in $servers) {
    $healthReport += Get-SystemHealth -ComputerName $server
}

# 4. Raporu Excel'e aktar
$healthReport | Export-Csv "C:\Reports\SystemHealth_$(Get-Date -Format 'yyyyMMdd').csv" -NoTypeInformation

🏢 Senaryo 3: Otomatik Backup

Problem: Kritik dosyaların manuel yedeklenmesi

Çözüm: PowerShell ile otomatik backup

# 1. Backup parametreleri
$sourcePath = "C:\CriticalData"
$backupPath = "D:\Backups\$(Get-Date -Format 'yyyyMMdd')"
$logPath = "C:\Logs\Backup.log"

# 2. Backup klasörü oluştur
if (!(Test-Path $backupPath)) {
    New-Item -ItemType Directory -Path $backupPath -Force
}

# 3. Backup işlemi
try {
    Write-Log -Message "Backup başlatıldı: $sourcePath -> $backupPath" -Level "INFO" -Path $logPath
    
    # Dosyaları kopyala
    Copy-Item -Path $sourcePath -Destination $backupPath -Recurse -Force
    
    # Backup boyutunu hesapla
    $backupSize = (Get-ChildItem -Path $backupPath -Recurse | Measure-Object -Property Length -Sum).Sum / 1GB
    
    Write-Log -Message "Backup tamamlandı. Boyut: $([math]::Round($backupSize, 2)) GB" -Level "INFO" -Path $logPath
}
catch {
    Write-Log -Message "Backup hatası: $($_.Exception.Message)" -Level "ERROR" -Path $logPath
}

Error Handling ve Logging

# Gelişmiş error handling
function Invoke-SafeCommand {
    param(
        [Parameter(Mandatory=$true)]
        [scriptblock]$Command,
        
        [Parameter(Mandatory=$true)]
        [string]$LogPath
    )
    
    try {
        $result = & $Command
        Write-Log -Message "Command executed successfully" -Level "INFO" -Path $LogPath
        return $result
    }
    catch {
        $errorMessage = "Command failed: $($_.Exception.Message)"
        Write-Log -Message $errorMessage -Level "ERROR" -Path $LogPath
        throw $_
    }
}

# Logging fonksiyonu
function Write-Log {
    param(
        [Parameter(Mandatory=$true)]
        [string]$Message,
        
        [Parameter(Mandatory=$true)]
        [ValidateSet("INFO", "WARNING", "ERROR")]
        [string]$Level,
        
        [Parameter(Mandatory=$true)]
        [string]$Path
    )
    
    $timestamp = Get-Date -Format "yyyy-MM-dd HH:mm:ss"
    $logEntry = "[$timestamp] [$Level] $Message"
    
    Add-Content -Path $Path -Value $logEntry
    
    if ($Level -eq "ERROR") {
        Write-Host $logEntry -ForegroundColor Red
    }
    elseif ($Level -eq "WARNING") {
        Write-Host $logEntry -ForegroundColor Yellow
    }
    else {
        Write-Host $logEntry -ForegroundColor Green
    }
}

// C# ile IT destek chatbot'u
using Microsoft.Bot.Builder;
using Microsoft.Bot.Schema;
using System.Threading;
using System.Threading.Tasks;

public class ITSupportBot : ActivityHandler
{
    protected override async Task OnMessageActivityAsync(ITurnContext turnContext, CancellationToken cancellationToken)
    {
        var userMessage = turnContext.Activity.Text.ToLower();
        
        string response = userMessage switch
        {
            "merhaba" => "Merhaba! IT destek chatbot'u burada. Size nasıl yardımcı olabilirim?",
            "şifre sıfırla" => "Şifre sıfırlama işlemi başlatılıyor. E-posta adresinizi girin:",
            "sistem durumu" => "Tüm sistemler çalışıyor. Son güncelleme: " + DateTime.Now.ToString("dd.MM.yyyy HH:mm"),
            "ticket oluştur" => "Ticket oluşturma işlemi başlatılıyor. Sorununuzu detaylı açıklayın:",
            "yardım" => "Size şu konularda yardımcı olabilirim:\n• Şifre sıfırlama\n• Sistem durumu\n• Ticket oluşturma\n• Yazılım desteği",
            _ => "Üzgünüm, anlamadım. 'yardım' yazarak desteklenen komutları görebilirsiniz."
        };
        
        await turnContext.SendActivityAsync(MessageFactory.Text(response), cancellationToken);
    }
}

# Python ile LUIS entegrasyonu - IT destek chatbot'u
import requests
import json

def get_luis_prediction(text, app_id, subscription_key, endpoint):
    url = f"{endpoint}/luis/prediction/v3.0/apps/{app_id}/slots/production/predict"
    params = {
        'query': text,
        'subscription-key': subscription_key
    }
    
    response = requests.get(url, params=params)
    return response.json()

# Gerçek hayat örneği
user_message = "Bilgisayarım çok yavaş çalışıyor, ne yapabilirim?"
prediction = get_luis_prediction(
    user_message, 
    "your-app-id", 
    "your-subscription-key", 
    "https://your-endpoint.cognitiveservices.azure.com"
)

intent = prediction['prediction']['topIntent']
entities = prediction['prediction']['entities']

# Intent'e göre yanıt ver
if intent == "PerformanceIssue":
    response = "Performans sorunu tespit edildi. Şu adımları deneyin:\n1. Bilgisayarı yeniden başlatın\n2. Gereksiz programları kapatın\n3. Disk temizliği yapın"
elif intent == "PasswordReset":
    response = "Şifre sıfırlama işlemi başlatılıyor. E-posta adresinizi girin:"
else:
    response = "Size nasıl yardımcı olabilirim? 'yardım' yazarak desteklenen komutları görebilirsiniz."

print(f"Kullanıcı: {user_message}")
print(f"AI Anlama: {intent}")
print(f"Chatbot Yanıtı: {response}")

🤖 QnA Maker - Bilgi Tabanı

Neden kullanılır: Sık sorulan soruları otomatik cevaplamak

# QnA Maker API kullanımı - IT bilgi tabanı
import requests
import json

def get_qna_answer(question, knowledge_base_id, subscription_key, endpoint):
    url = f"{endpoint}/knowledgebases/{knowledge_base_id}/generateAnswer"
    headers = {
        'Authorization': f'EndpointKey {subscription_key}',
        'Content-Type': 'application/json'
    }
    data = {
        'question': question
    }
    
    response = requests.post(url, headers=headers, json=data)
    return response.json()

# Gerçek hayat örneği
user_question = "RDP bağlantısı nasıl kurulur?"
answer = get_qna_answer(
    user_question,
    "your-kb-id",
    "your-subscription-key",
    "https://your-endpoint.cognitiveservices.azure.com"
)

if answer['answers'] and answer['answers'][0]['score'] > 0.7:
    response = answer['answers'][0]['answer']
    print(f"Soru: {user_question}")
    print(f"AI Yanıtı: {response}")
else:
    print("Üzgünüm, bu soruya yanıt veremiyorum. Lütfen IT destek ekibiyle iletişime geçin.")

🔒 Güvenlik

Neden Önemli:

Otomasyon güvenlik açıkları yaratabilir. Güvenlik öncelikli olmalıdır.

Pratik Örnekler:

  • Least Privilege: Minimum gerekli yetki
  • Credential Storage: Güvenli şifre saklama
  • Audit Logging: Tüm işlemleri logla
  • Access Controls: Erişim kontrolleri
  • Security Reviews: Düzenli güvenlik incelemeleri
Güvenlik Örneği:
# Güvenli PowerShell script
$securePassword = ConvertTo-SecureString -String $password -AsPlainText -Force
$credential = New-Object System.Management.Automation.PSCredential($username, $securePassword)

# Sadece gerekli yetkilerle çalış
Invoke-Command -ComputerName $server -Credential $credential -ScriptBlock {
    # Sadece gerekli işlemler
}

  • Usage Analytics: Kullanım analizi
  • Continuous Improvement: Sürekli iyileştirme
İzleme Örneği:
# PowerShell ile otomasyon izleme
$automationMetrics = @{
    StartTime = Get-Date
    SuccessCount = 0
    ErrorCount = 0
    TotalProcessed = 0
}

# Her işlem sonrası metrikleri güncelle
if ($success) {
    $automationMetrics.SuccessCount++
} else {
    $automationMetrics.ErrorCount++
}

# Günlük rapor oluştur
$automationMetrics | Export-Csv "C:\Reports\AutomationMetrics_$(Get-Date -Format 'yyyyMMdd').csv"
