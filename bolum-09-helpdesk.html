# 1. Otomatik ticket atama
function Auto-AssignTicket {
    param($ticket)
    
    $category = $ticket.Category
    $priority = $ticket.Priority
    
    switch ($category) {
        "Network" { $assignee = "network.team@company.com" }
        "Security" { $assignee = "security.team@company.com" }
        "Hardware" { $assignee = "hardware.team@company.com" }

        default { $assignee = "general.support@company.com" }
    }
    
    Assign-Ticket -TicketId $ticket.Id -Assignee $assignee
    Send-Notification -To $assignee -Message "New ticket assigned: $($ticket.Title)"
}

function Monitor-SLA {
    $overdueTickets = Get-Tickets -Status "Open" | 
        Where-Object { $_.Created -lt (Get-Date).AddHours(-$_.SLA) }
    
    foreach ($ticket in $overdueTickets) {
        Escalate-Ticket -TicketId $ticket.Id -Reason "SLA exceeded"
        Send-Alert -To "manager@company.com" -Message "SLA exceeded for ticket: $($ticket.Id)"
    }
}
